solace_queue_cancel_replay -- cancel replay on queue
====================================================

Cancel replay on a Queue object.




Examples
--------

.. code-block:: yaml+jinja

   
   hosts: all
   gather_facts: no
   any_errors_fatal: true
   collections:
   - solace.pubsub_plus
   module_defaults:
       solace_queue:
         host: "{{ sempv2_host }}"
         port: "{{ sempv2_port }}"
         secure_connection: "{{ sempv2_is_secure_connection }}"
         username: "{{ sempv2_username }}"
         password: "{{ sempv2_password }}"
         timeout: "{{ sempv2_timeout }}"
         msg_vpn: "{{ vpn }}"
       solace_queue_subscription:
         host: "{{ sempv2_host }}"
         port: "{{ sempv2_port }}"
         secure_connection: "{{ sempv2_is_secure_connection }}"
         username: "{{ sempv2_username }}"
         password: "{{ sempv2_password }}"
         timeout: "{{ sempv2_timeout }}"
         msg_vpn: "{{ vpn }}"
       solace_queue_start_replay:
         host: "{{ sempv2_host }}"
         port: "{{ sempv2_port }}"
         secure_connection: "{{ sempv2_is_secure_connection }}"
         username: "{{ sempv2_username }}"
         password: "{{ sempv2_password }}"
         timeout: "{{ sempv2_timeout }}"
         msg_vpn: "{{ vpn }}"
       solace_queue_cancel_replay:
         host: "{{ sempv2_host }}"
         port: "{{ sempv2_port }}"
         secure_connection: "{{ sempv2_is_secure_connection }}"
         username: "{{ sempv2_username }}"
         password: "{{ sempv2_password }}"
         timeout: "{{ sempv2_timeout }}"
         msg_vpn: "{{ vpn }}"
   tasks:
   - name: create queue
     solace_queue:
       name: bar
       state: present

   - name: add queue subscription
     solace_queue_subscription:
       name: "hello/world"
       queue_name: bar
       state: present

   - name: start replay
     solace_queue_start_replay:
       queue_name: bar
       settings:
           fromTime: 1

   - name: cancel replay
     solace_queue_cancel_replay:
       queue_name: bar
       settings:
           forceComplete: true
